import React, { useState, useEffect } from "react";

import { View, Text, TouchableOpacity, StyleSheet, ScrollView } from "react-native";

import { Audio } from "expo-av";


const notes = [

{ note: "C4", label: "C", color: "white" },

{ note: "Cs4", label: "C#", color: "black", position: 1 },

{ note: "D4", label: "D", color: "white" },

{ note: "Ds4", label: "D#", color: "black", position: 2 },

{ note: "E4", label: "E", color: "white" },

{ note: "F4", label: "F", color: "white" },

{ note: "Fs4", label: "F#", color: "black", position: 4 },

{ note: "G4", label: "G", color: "white" },

{ note: "Gs4", label: "G#", color: "black", position: 5 },

{ note: "A4", label: "A", color: "white" },

{ note: "As4", label: "A#", color: "black", position: 6 },

{ note: "B4", label: "B", color: "white" },

];


export default function App() {

const [pressedKey, setPressedKey] = useState(null);


async function playSound(note) {

const soundObject = new Audio.Sound();

try {

const path = require(`./assets/sounds/${note}.mp3`);

await soundObject.loadAsync(path);

await soundObject.playAsync();

setPressedKey(note);

setTimeout(() => setPressedKey(null), 200);

} catch (error) {

console.log("Error al reproducir sonido:", error);

}

}


return (

<View style={styles.container}>

<Text style={styles.title}>ðŸŽ¹ Piano App</Text>

<ScrollView

horizontal

showsHorizontalScrollIndicator={false}

contentContainerStyle={styles.keyboard}

>

{/* Teclas blancas */}

{notes

.filter((n) => n.color === "white")

.map((n, i) => (

<TouchableOpacity

key={n.note}

onPress={() => playSound(n.note)}

style={[

styles.whiteKey,

pressedKey === n.note && { backgroundColor: "#ddd" },

]}

>

<Text style={styles.whiteLabel}>{n.label}</Text>

</TouchableOpacity>

))}


{/* Teclas negras */}

{notes

.filter((n) => n.color === "black")

.map((n) => (

<TouchableOpacity

key={n.note}

onPress={() => playSound(n.note)}

style={[

styles.blackKey,

styles[`pos${n.position}`],

pressedKey === n.note && { backgroundColor: "#555" },

]}

>

<Text style={styles.blackLabel}>{n.label}</Text>

</TouchableOpacity>

))}

</ScrollView>

</View>

);

}


const styles = StyleSheet.create({

container: {

flex: 1,

backgroundColor: "#111",

alignItems: "center",

justifyContent: "center",

},

title: {

color: "#fff",

fontSize: 28,

},

keyboard: {

},

width: 60,

borderWidth: 1,

justifyContent: "flex-end",

alignItems: "center",

marginRight: 2,

},

whiteLabel: {

color: "#333",

marginBottom: 10,

},

blackKey: {

position: "absolute",

width: 40,

height: 120,

backgroundColor: "#000",

zIndex: 1,

marginLeft: 45,

borderRadius: 4,

alignItems: "center",

justifyContent: "flex-end",

},

blackLabel: {

color: "#fff",

fontSize: 10,

marginBottom: 8,

},

pos1: { left: 40 },

pos2: { left: 100 },

pos4: { left: 220 },

pos5: { left: 280 },

pos6: { left: 340 },

});
